public class reverse{
 static void reversearray(int a[]){
   int i=0,j=a.length-1;
   while(i<j){
     int temp = a[i];
     a[i]= a[j];
     a[j]= temp;;
     i++;
     j--;
    }
   }
  public static void main(String args[]){
   int a[] = {1,2,3,4,5};
   for(int i=0;i<=a.length-1;i++){
   System.out.print(a[i] + " ");
   }
   reversearray(a);
   System.out.println();
   for(int i=0;i<=a.length-1;i++){
   System.out.print(a[i] + " ");
   }
 }
}
